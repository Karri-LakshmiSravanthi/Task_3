document.addEventListener("DOMContentLoaded", function () {
    document.getElementById("searchInput").addEventListener("input", function () {
      var input, filter, dropdown, options, i, txtValue;
      input = this.value;
      filter = input.toUpperCase();
      dropdown = document.getElementById("dropdownContent");
      options = dropdown.getElementsByClassName("option");
      if (input.length > 0) {
        dropdown.classList.add("show");
      } else {
        dropdown.classList.remove("show");
      }
      for (i = 0; i < options.length; i++) {
        txtValue = options[i].getElementsByClassName("name")[0].textContent || options[i].getElementsByClassName("name")[0].innerText;
        if (txtValue.toUpperCase().indexOf(filter) > -1) {
          options[i].style.display = "";
        } else {
          options[i].style.display = "none";
        }
      }
    });
  
    var checkboxes = document.querySelectorAll('.option-checkbox');
    checkboxes.forEach(function (checkbox) {
      checkbox.addEventListener('change', function () {
        if (this.checked) {
          // Uncheck previously checked checkboxes
          checkboxes.forEach(function (otherCheckbox) {
            if (otherCheckbox !== checkbox) {
              otherCheckbox.checked = false;
            }
          });
          var name = this.closest('.option').querySelector('.name').textContent;
          document.getElementById('searchInput').value = name;
        } else {
          document.getElementById('searchInput').value = "";
        }
      });
    });
